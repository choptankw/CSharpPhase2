using System;
using System.Linq;

namespace StringReverseException
{
    class Program
    {
        static void Main(string[] args)
        {
            string string1 = Console.ReadLine();

            foreach (char a in string1)
            {
                if (Char.IsDigit(a))
                {
                    throw new Exception("There are numbers in string");
                }
            }

            Console.WriteLine(ReverseString(string1));  
        }

        static string ReverseString(string _string)
        {
            string reversedString = String.Empty;
            for(int i = _string.Length - 1; i >= 0 ; i--)
            {
                reversedString += _string[i];
            }

            return reversedString;
        }
    }
}
