using System;
using System.IO;

namespace Fly
{
    class Program
    {
        static void Main(string[] args)
        {
            try
            {
                string[] arr1 = { "1 Line", "2 Line", "3 Line" };
                WriteLinesInFile(arr1);
            }
            catch(FormatException)
            {
                Console.WriteLine("Wrong input");
            }
        }

        static void WriteLinesInFile(string[] array)
        {
            File.WriteAllLines("D:\\File1.txt", array);
        }
    }
}
