using System;
using System.Collections.Generic;

namespace List
{
    class Program
    {
        static void Main(string[] args)
        {
            List<int> List = new List<int>(4);

            List.Add(5);
            List.Add(10);
            List.Add(15);
            List.Add(20);

            Console.WriteLine(ReturnIndexItem(List, 2));
        }

        static T ReturnIndexItem<T>(List<T> list, int index)
        {
            return list[index];
        }
    }
}
