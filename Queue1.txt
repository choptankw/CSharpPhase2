using System;
using System.Linq;
using System.Collections.Generic;

namespace Queue
{
    class Program
    {
        static void Main(string[] args)
        {
            Queue<string> MyQ = new Queue<string>();
            MyQ.Enqueue("One,");
            MyQ.Enqueue("Two,");
            MyQ.Enqueue("Three,");
            MyQ.Enqueue("Four");
            MyQ.Dequeue();

            PrintQueue(MyQ);
        }

        static void PrintQueue<T>(Queue<T> queue)
        {
            foreach(T item in queue)
            {
                Console.WriteLine($" {item}");
            }
        }
    }
}
