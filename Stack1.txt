using System;
using System.Collections.Generic;

namespace Stack
{
    class Program
    {
        static void Main(string[] args)
        {
            Stack<int> stack = new Stack<int>();
            stack.Push(5);
            stack.Push(10);
            stack.Push(15);
            stack.Push(20);
            stack.Push(25);

            PrintStack(stack);
        }

        static void PrintStack<T>(Stack<T> stack)
        {
            foreach(object item in stack)
            {
                Console.WriteLine(item);
            }
        }
    }
}
